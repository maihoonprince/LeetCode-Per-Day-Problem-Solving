March-5.Minimum Length of String After Deleting Similar Ends:
1750.

class Solution {
    public int minimumLength(String s) {
        int l=0, r=s.length()-1;
        while(l<r && s.charAt(l) == s.charAt(r)){
            char ch = s.charAt(l);
            while(l <= r && s.charAt(l) == ch)
            l++;

            while(r>= l && s.charAt(r) == ch)
            r--;
        }
        return l>r? 0 : r-l+1;
    }
}
